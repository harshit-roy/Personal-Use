CREATE OR REPLACE PROCEDURE EVOLVE.GET_CASE_HISTORY_FILTER_METADATA
(
    P_CASE_ID        IN  EVOLVE.CASE_CLAIMS.CASE_ID%TYPE,
    P_OUTPUT_CURSOR  OUT SYS_REFCURSOR
)
AS
BEGIN
    OPEN P_OUTPUT_CURSOR FOR
        WITH
        EMP_BASE_LOG AS
        (
            SELECT
                NVL(L.MODIFIED_DATE, L.CREATED_DATE) AS EVENT_DATE,
                NVL(L.MODIFIED_BY,   L.CREATED_BY)   AS EVENT_USER,
                TO_CHAR(L.EMPR_NAME)                   AS EMPR_NAME,
                TO_CHAR(L.EMPR_INFO_CD)                AS EMPR_INFO_CD,
                TO_CHAR(L.EMPR_TIN)                    AS EMPR_TIN,
                TO_CHAR(L.GROUP_SIZE)                  AS GROUP_SIZE,
                TO_CHAR(L.EMPR_ADDR1)                  AS EMPR_ADDR1,
                TO_CHAR(L.EMPR_ADDR2)                  AS EMPR_ADDR2,
                TO_CHAR(L.EMPR_CITY)                   AS EMPR_CITY,
                TO_CHAR(L.EMPR_STATE)                  AS EMPR_STATE,
                TO_CHAR(L.EMPR_ZIP)                    AS EMPR_ZIP,
                TO_CHAR(L.EMPR_PHONE_NUMBER)           AS EMPR_PHONE_NUMBER,
                TO_CHAR(L.EMPR_FAX)                    AS EMPR_FAX,
                TO_CHAR(L.EMPR_EMAIL)                  AS EMPR_EMAIL,
                TO_CHAR(L.EMPLOYER_PREFERRED_DELIVERY) AS EMPLOYER_PREFERRED_DELIVERY,

                LAG(TO_CHAR(L.EMPR_NAME)) OVER
                (
                    PARTITION BY L.CASE_CLAIM_ID
                    ORDER BY NVL(L.MODIFIED_DATE, L.CREATED_DATE), L.CASE_CLAIM_LOG_ID
                ) AS PREV_EMPR_NAME,

                LAG(TO_CHAR(L.EMPR_INFO_CD)) OVER
                (
                    PARTITION BY L.CASE_CLAIM_ID
                    ORDER BY NVL(L.MODIFIED_DATE, L.CREATED_DATE), L.CASE_CLAIM_LOG_ID
                ) AS PREV_EMPR_INFO_CD,

                LAG(TO_CHAR(L.EMPR_TIN)) OVER
                (
                    PARTITION BY L.CASE_CLAIM_ID
                    ORDER BY NVL(L.MODIFIED_DATE, L.CREATED_DATE), L.CASE_CLAIM_LOG_ID
                ) AS PREV_EMPR_TIN,

                LAG(TO_CHAR(L.GROUP_SIZE)) OVER
                (
                    PARTITION BY L.CASE_CLAIM_ID
                    ORDER BY NVL(L.MODIFIED_DATE, L.CREATED_DATE), L.CASE_CLAIM_LOG_ID
                ) AS PREV_GROUP_SIZE,

                LAG(TO_CHAR(L.EMPR_ADDR1)) OVER
                (
                    PARTITION BY L.CASE_CLAIM_ID
                    ORDER BY NVL(L.MODIFIED_DATE, L.CREATED_DATE), L.CASE_CLAIM_LOG_ID
                ) AS PREV_EMPR_ADDR1,

                LAG(TO_CHAR(L.EMPR_ADDR2)) OVER
                (
                    PARTITION BY L.CASE_CLAIM_ID
                    ORDER BY NVL(L.MODIFIED_DATE, L.CREATED_DATE), L.CASE_CLAIM_LOG_ID
                ) AS PREV_EMPR_ADDR2,

                LAG(TO_CHAR(L.EMPR_CITY)) OVER
                (
                    PARTITION BY L.CASE_CLAIM_ID
                    ORDER BY NVL(L.MODIFIED_DATE, L.CREATED_DATE), L.CASE_CLAIM_LOG_ID
                ) AS PREV_EMPR_CITY,

                LAG(TO_CHAR(L.EMPR_STATE)) OVER
                (
                    PARTITION BY L.CASE_CLAIM_ID
                    ORDER BY NVL(L.MODIFIED_DATE, L.CREATED_DATE), L.CASE_CLAIM_LOG_ID
                ) AS PREV_EMPR_STATE,

                LAG(TO_CHAR(L.EMPR_ZIP)) OVER
                (
                    PARTITION BY L.CASE_CLAIM_ID
                    ORDER BY NVL(L.MODIFIED_DATE, L.CREATED_DATE), L.CASE_CLAIM_LOG_ID
                ) AS PREV_EMPR_ZIP,

                LAG(TO_CHAR(L.EMPR_PHONE_NUMBER)) OVER
                (
                    PARTITION BY L.CASE_CLAIM_ID
                    ORDER BY NVL(L.MODIFIED_DATE, L.CREATED_DATE), L.CASE_CLAIM_LOG_ID
                ) AS PREV_EMPR_PHONE_NUMBER,

                LAG(TO_CHAR(L.EMPR_FAX)) OVER
                (
                    PARTITION BY L.CASE_CLAIM_ID
                    ORDER BY NVL(L.MODIFIED_DATE, L.CREATED_DATE), L.CASE_CLAIM_LOG_ID
                ) AS PREV_EMPR_FAX,

                LAG(TO_CHAR(L.EMPR_EMAIL)) OVER
                (
                    PARTITION BY L.CASE_CLAIM_ID
                    ORDER BY NVL(L.MODIFIED_DATE, L.CREATED_DATE), L.CASE_CLAIM_LOG_ID
                ) AS PREV_EMPR_EMAIL,

                LAG(TO_CHAR(L.EMPLOYER_PREFERRED_DELIVERY)) OVER
                (
                    PARTITION BY L.CASE_CLAIM_ID
                    ORDER BY NVL(L.MODIFIED_DATE, L.CREATED_DATE), L.CASE_CLAIM_LOG_ID
                ) AS PREV_EMPLOYER_PREFERRED_DELIVERY
            FROM EVOLVE.CASE_CLAIMS_LOG L
            WHERE L.CASE_ID = P_CASE_ID
        ),

        EMPLOYER_HISTORY_EVENTS AS
        (
            SELECT DISTINCT
                TRUNC(EVENT_DATE) AS EVENT_DATE,
                EVENT_USER,
                'Employer' AS CONTAINER
            FROM EMP_BASE_LOG
            WHERE
                    (PREV_EMPR_NAME IS NULL AND EMPR_NAME IS NOT NULL)
                 OR (PREV_EMPR_NAME IS NOT NULL AND EMPR_NAME IS NULL)
                 OR (PREV_EMPR_NAME <> EMPR_NAME)
                 OR (PREV_EMPR_INFO_CD <> EMPR_INFO_CD)
                 OR (PREV_EMPR_TIN <> EMPR_TIN)
                 OR (PREV_GROUP_SIZE <> GROUP_SIZE)
                 OR (PREV_EMPR_ADDR1 <> EMPR_ADDR1)
                 OR (PREV_EMPR_ADDR2 <> EMPR_ADDR2)
                 OR (PREV_EMPR_CITY <> EMPR_CITY)
                 OR (PREV_EMPR_STATE <> EMPR_STATE)
                 OR (PREV_EMPR_ZIP <> EMPR_ZIP)
                 OR (PREV_EMPR_PHONE_NUMBER <> EMPR_PHONE_NUMBER)
                 OR (PREV_EMPR_FAX <> EMPR_FAX)
                 OR (PREV_EMPR_EMAIL <> EMPR_EMAIL)
                 OR (PREV_EMPLOYER_PREFERRED_DELIVERY <> EMPLOYER_PREFERRED_DELIVERY)
        ),

        ATT_BASE_LOG AS
        (
            SELECT
                L.ACTION_DATE,
                L.ACTION_BY,
                L.ACTION_TYPE,
                L.OLD_CASE_ID,
                L.NEW_CASE_ID,
                L.OLD_CASE_CLAIM_ID,
                L.NEW_CASE_CLAIM_ID,
                TO_CHAR(L.OLD_DOCUMENT_NAME)        AS OLD_DOCUMENT_NAME,
                TO_CHAR(L.NEW_DOCUMENT_NAME)        AS NEW_DOCUMENT_NAME,
                TO_CHAR(L.OLD_DOCUMENT_DESCRIPTION) AS OLD_DOCUMENT_DESCRIPTION,
                TO_CHAR(L.NEW_DOCUMENT_DESCRIPTION) AS NEW_DOCUMENT_DESCRIPTION,
                TO_CHAR(L.OLD_DOCUMENT_LINK)        AS OLD_DOCUMENT_LINK,
                TO_CHAR(L.NEW_DOCUMENT_LINK)        AS NEW_DOCUMENT_LINK,
                TO_CHAR(L.OLD_DOCUMENT_VERSION)     AS OLD_DOCUMENT_VERSION,
                TO_CHAR(L.NEW_DOCUMENT_VERSION)     AS NEW_DOCUMENT_VERSION,
                TO_CHAR(L.OLD_DOCUMENT_CATEGORY)    AS OLD_DOCUMENT_CATEGORY,
                TO_CHAR(L.NEW_DOCUMENT_CATEGORY)    AS NEW_DOCUMENT_CATEGORY,
                TO_CHAR(L.OLD_DOCUMENT_TYPE)        AS OLD_DOCUMENT_TYPE,
                TO_CHAR(L.NEW_DOCUMENT_TYPE)        AS NEW_DOCUMENT_TYPE,
                TO_CHAR(L.OLD_IS_ACTIVE)            AS OLD_IS_ACTIVE,
                TO_CHAR(L.NEW_IS_ACTIVE)            AS NEW_IS_ACTIVE,
                TO_CHAR(L.OLD_DELIVERY_METHOD)      AS OLD_DELIVERY_METHOD,
                TO_CHAR(L.NEW_DELIVERY_METHOD)      AS NEW_DELIVERY_METHOD
            FROM EVOLVE.CASE_DOCUMENTS_LOG L
            WHERE L.OLD_CASE_ID = P_CASE_ID
               OR L.NEW_CASE_ID = P_CASE_ID
        ),

        ATTACHMENT_HISTORY_EVENTS AS
        (
            SELECT DISTINCT
                TRUNC(ACTION_DATE) AS EVENT_DATE,
                ACTION_BY          AS EVENT_USER,
                'Attachments'      AS CONTAINER
            FROM ATT_BASE_LOG
            WHERE
                   ACTION_TYPE = 'I'
                OR ACTION_TYPE = 'D'
                OR (
                       ACTION_TYPE = 'U'
                   AND OLD_IS_ACTIVE = 'Y'
                   AND NEW_IS_ACTIVE = 'N'
                   AND NVL(OLD_CASE_ID, -1) = NVL(NEW_CASE_ID, -1)
                   AND NVL(OLD_CASE_CLAIM_ID, -1) = NVL(NEW_CASE_CLAIM_ID, -1)
                   )
                OR (
                       ACTION_TYPE = 'U'
                   AND (
                          NVL(OLD_CASE_ID, -1) <> NVL(NEW_CASE_ID, -1)
                       OR NVL(OLD_CASE_CLAIM_ID, -1) <> NVL(NEW_CASE_CLAIM_ID, -1)
                       )
                   )
                OR (
                       ACTION_TYPE = 'U'
                   AND NVL(OLD_CASE_ID, -1) = NVL(NEW_CASE_ID, -1)
                   AND NVL(OLD_CASE_CLAIM_ID, -1) = NVL(NEW_CASE_CLAIM_ID, -1)
                   AND NVL(OLD_IS_ACTIVE, '#') = NVL(NEW_IS_ACTIVE, '#')
                   AND (
                          (OLD_DOCUMENT_NAME IS NOT NULL AND NEW_DOCUMENT_NAME IS NOT NULL AND OLD_DOCUMENT_NAME <> NEW_DOCUMENT_NAME)
                       OR (OLD_DOCUMENT_DESCRIPTION IS NOT NULL AND NEW_DOCUMENT_DESCRIPTION IS NOT NULL AND OLD_DOCUMENT_DESCRIPTION <> NEW_DOCUMENT_DESCRIPTION)
                       OR (OLD_DOCUMENT_LINK IS NOT NULL AND NEW_DOCUMENT_LINK IS NOT NULL AND OLD_DOCUMENT_LINK <> NEW_DOCUMENT_LINK)
                       OR (OLD_DOCUMENT_VERSION IS NOT NULL AND NEW_DOCUMENT_VERSION IS NOT NULL AND OLD_DOCUMENT_VERSION <> NEW_DOCUMENT_VERSION)
                       OR (OLD_DOCUMENT_CATEGORY IS NOT NULL AND NEW_DOCUMENT_CATEGORY IS NOT NULL AND OLD_DOCUMENT_CATEGORY <> NEW_DOCUMENT_CATEGORY)
                       OR (OLD_DOCUMENT_TYPE IS NOT NULL AND NEW_DOCUMENT_TYPE IS NOT NULL AND OLD_DOCUMENT_TYPE <> NEW_DOCUMENT_TYPE)
                       OR (OLD_DELIVERY_METHOD IS NOT NULL AND NEW_DELIVERY_METHOD IS NOT NULL AND OLD_DELIVERY_METHOD <> NEW_DELIVERY_METHOD)
                       )
                   )
        ),

        ALL_HISTORY_EVENTS AS
        (
            SELECT EVENT_DATE, EVENT_USER, CONTAINER
            FROM EMPLOYER_HISTORY_EVENTS
            UNION ALL
            SELECT EVENT_DATE, EVENT_USER, CONTAINER
            FROM ATTACHMENT_HISTORY_EVENTS
        )

        SELECT
            'Event_Date' AS FILTER_TYPE,
            TO_CHAR(EVENT_DATE, 'YYYY-MM-DD') AS FILTER_VALUE
        FROM
        (
            SELECT DISTINCT EVENT_DATE
            FROM ALL_HISTORY_EVENTS
            WHERE EVENT_DATE IS NOT NULL
        )

        UNION ALL

        SELECT
            'Container' AS FILTER_TYPE,
            CONTAINER   AS FILTER_VALUE
        FROM
        (
            SELECT DISTINCT CONTAINER
            FROM ALL_HISTORY_EVENTS
            WHERE CONTAINER IS NOT NULL
        )

        UNION ALL

        SELECT DISTINCT
            'User_Name' AS FILTER_TYPE,
            COALESCE
            (
                TO_CHAR(AD.AUDITOR_NAME),
                TO_CHAR(U.EVENT_USER),
                'SYSTEM'
            ) AS FILTER_VALUE
        FROM
        (
            SELECT DISTINCT EVENT_USER
            FROM ALL_HISTORY_EVENTS
        ) U
        LEFT JOIN EVOLVE.AUDITOR_DETAILS AD
               ON TO_CHAR(AD.AUDITOR_USER_ID) = TO_CHAR(U.EVENT_USER)
        WHERE COALESCE(TO_CHAR(AD.AUDITOR_NAME), TO_CHAR(U.EVENT_USER), 'SYSTEM') IS NOT NULL;

END;
